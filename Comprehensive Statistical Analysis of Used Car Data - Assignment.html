#categorizing fuel type as CNG, Diesel, LPG, Petrol

cng = df[df['fuel']=='CNG']['selling_price']
diesel = df[df['fuel']=='Diesel']['selling_price']
lpg = df[df['fuel']=='LPG']['selling_price']
petrol = df[df['fuel']=='Petrol']['selling_price']

#performing oneway ANOVA Test
print('One-Way ANOVA Test:')
result=stats.f_oneway(cng,diesel,lpg,petrol)
print("f-statistic:",result.statistic)
print("p-Value:",result.pvalue)

#inferences  --declaring 0.05 as the alpha.
if (result.pvalue < 0.05):
    print("Reject the null hypothesis. There is a significant difference in the selling price of the car based on the fuel type.")
else:
    print("Failed to reject the null hypothesis. There is no significant difference in the selling price of the car based on the fuel type.")    

# identifying which fuel type has highest and lowest selling price
high_sp = mean_prices.idxmax()
low_sp = mean_prices.idxmin()
