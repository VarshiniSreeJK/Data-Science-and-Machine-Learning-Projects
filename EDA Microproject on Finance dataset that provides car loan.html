#Converting 'Amount_Requested_for_loan', 'Amount_Funded_By_Investors', 'Family_income' to float

#Removing unwanted characters from 'credit_score' and converting it to int data type

def missing_values(df):
    missing_count = df.isnull().sum()
    missing_percent = (missing_count / len(df)) * 100
    missing_info = pd.DataFrame({'Missing_Count': missing_count, 'Missing_Percentage': missing_percent})
    return missing_info

def cat_col(df, feature):
    df[feature] = df[feature].fillna(df[feature].mode()[0])
def num_col(df, feature):
    df[feature] = df[feature].fillna(df[feature].mean())
